## Approach
Todays daily is quite straight forward, just do level search and sum up the value for each level, and see which is the maximum.

## Code
```c++
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        vector<int> sum;
        traverse(sum, root, 0);
        return max_element(sum.begin(), sum.end()) - sum.begin() + 1;
    }

    void traverse(vector<int>& sum, TreeNode* root, int level) {
        if(level >= (int)sum.size()) sum.resize(sum.size() + 1);
        sum[level] += root->val;
        if(root->left) traverse(sum, root->left, level+1);
        if(root->right) traverse(sum, root->right, level+1);
    }
};
```
